```markdown
[תיאור קצר של המסמך]

## פירוט

```

### 3. תיעוד קוד
```typescript
/**
 * תיאור הפונקציה
 * @param param1 - תיאור הפרמטר
 * @param param2 - תיאור הפרמטר
 * @returns - תיאור הערך המוחזר
 * @throws - תיאור השגיאות האפשריות
 * @example
 * // דוגמת שימוש
 * const result = myFunction('test', 123);
 */
function myFunction(param1: string, param2: number): Result {
    // הערות מסבירות בתוך הקוד
    const result = doSomething(param1, param2);
    return result;
}
```
